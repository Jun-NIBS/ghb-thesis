\begingroup%
\figureFont\figureFontSize%
\pgfdeclarelayer{background1}%
\pgfdeclarelayer{background2}%
\pgfsetlayers{background2,background1,main}%
\begin{tikzpicture}[%
    value node/.style={%
      op node,
    },
    outer match node/.style={%
      match node,
      draw=none,
      inner sep=0,
    },
    match label/.style={%
      minimum size=0,
      inner sep=0,
      outer sep=1pt,
      node distance=10pt,
      font=\footnotesize,
    },
  ]

  % Trees
  \node [op node] (add) {\opAdd};
  \node [value node, position=-135 degrees from add] (a) {\opVar{a}};
  \node [value node, position=- 45 degrees from add] (b) {\opVar{b}};
  \begin{scope}[data flow]
    \draw (a) -- (add);
    \draw (b) -- (add);
  \end{scope}

  \node [value node, right=of b] (c) {\opVar{c}};
  \node [op node, position=45 degrees from c] (mul) {\opMul};
  \node [value node, position=- 45 degrees from mul] (t1) {\opVar{t}};
  \begin{scope}[data flow]
    \draw (c) -- (mul);
    \draw (t1) -- (mul);
  \end{scope}

  \node [value node, right=of t1] (t2) {\opVar{t}};
  \node [op node, above=of t2] (load) {\opLoad};
  \begin{scope}[data flow]
    \draw (t2) -- (load);
  \end{scope}

  % Matches
  \begin{pgfonlayer}{background1}
    \foreach \i/\m/\p in {add/1/left, mul/2/left, load/4/right} {
      \node [match node, fill=shade2, fit=(\i)] (m\m) {};
      \node [match label, \p=of m\m] (m\m l) {$\strut m_{\m}$};
      \draw [thick] (m\m) -- (m\m l);
    }
  \end{pgfonlayer}
\end{tikzpicture}%
\endgroup%
