\input{figures/constraint-programming/square-styles}%
\begin{tikzpicture}[
    square/.append style={
      draw=none,
    },
  ]

\tikzset{}

  \node [square, minimum size=3*\squareSize] (s1) {};
  \node [square, minimum size=3*\squareSize, below right=0 and 5mm of s1.west]
        (s2) {};

  \def\sIPath{%
    (s1.north east) -- (s1.south east) -- (s1.south west) -- (s1.north west)
    -- cycle%
  }
  \def\sIIPath{%
    (s2.north east) -- (s2.south east) -- (s2.south west) -- (s2.north west)
    -- cycle%
  }
  \draw [boundary] \sIPath;
  \draw [boundary] \sIIPath;
\end{tikzpicture}%
\endgroup%
